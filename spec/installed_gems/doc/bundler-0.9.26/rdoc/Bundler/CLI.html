 79:       opts[:without] ||= []

 81: 
 82:       # Can't use Bundler.settings for this because settings needs gemfile.dirname
 83:       ENV['BUNDLE_GEMFILE'] = opts[:gemfile] if opts[:gemfile]
 84:       Bundler.settings[:path] = path if path
 85:       Bundler.settings[:disable_shared_gems] = '1' if options["disable-shared-gems"]
 86:       Bundler.settings.without = opts[:without]
 87: 
 88:       remove_lockfiles if options[:relock]
 89: 
 90:       begin
 91:         Installer.install(Bundler.root, Bundler.definition, opts)
 92:       rescue GemfileChanged
 93:         raise GemfileChanged, "You changed your Gemfile after locking. " +
 94:           "Please run `bundle install --relock`."
 95:       end
 96: 
 97:       lock if options[:relock]
 98:       cache if Bundler.root.join("vendor/cache").exist? && !options[:no_cache]

100:         "Use `bundle show [gemname]` to see where a bundled gem is installed."
