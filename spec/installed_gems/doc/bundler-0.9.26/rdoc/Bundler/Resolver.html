 78:     def resolve(reqs, activated)
 79:       # If the requirements are empty, then we are in a success state. Aka, all
 80:       # gem dependencies have been resolved.
 81:       throw :success, activated if reqs.empty?
 82: 
 83:       debug { print "\e[2J\e[f" ; "==== Iterating ====\n\n" }
 84: 
 85:       # Sort dependencies so that the ones that are easiest to resolve are first.
 86:       # Easiest to resolve is defined by:
 87:       #   1) Is this gem already activated?
 88:       #   2) Do the version requirements include prereleased gems?
 89:       #   3) Sort by number of gems available in the source.
 90:       reqs = reqs.sort_by do |a|
 91:         [ activated[a.name] ? 0 : 1,
 92:           a.requirement.prerelease? ? 0 : 1,
 93:           @errors[a.name]   ? 0 : 1,
 94:           activated[a.name] ? 0 : search(a).size ]
 95:       end
 96: 
 97:       debug { "Activated:\n" + activated.values.map { |a| "  #{a.name} (#{a.version})" }.join("\n") }
 98:       debug { "Requirements:\n" + reqs.map { |r| "  #{r.name} (#{r.requirement})"}.join("\n") }
 99: 
100:       activated = activated.dup
101: 
102:       if reqs.first.name == "bundler" && !activated["bundler"]
103:         # activate the current version of bundler before other versions
104:         bundler_version = ENV["BUNDLER_VERSION"] || Bundler::VERSION
105:         current = Gem::Dependency.new("bundler", bundler_version, reqs.first.type)
106:       else
107:         # Pull off the first requirement so that we can resolve it
108:         current = reqs.shift
109:       end
110: 
111:       debug { "Attempting:\n  #{current.name} (#{current.requirement})"}
112: 
113:       # Check if the gem has already been activated, if it has, we will make sure
114:       # that the currently activated gem satisfies the requirement.
115:       if existing = activated[current.name]
116:         if current.requirement.satisfied_by?(existing.version)
117:           debug { "    * [SUCCESS] Already activated" }
118:           @errors.delete(existing.name)
119:           # Since the current requirement is satisfied, we can continue resolving
120:           # the remaining requirements.
121:           resolve(reqs, activated)
122:         else
123:           debug { "    * [FAIL] Already activated" }
124:           @errors[existing.name] = [existing, current]
125:           debug { current.required_by.map {|d| "      * #{d.name} (#{d.requirement})" }.join("\n") }
126:           # debug { "    * All current conflicts:\n" + @errors.keys.map { |c| "      - #{c}" }.join("\n") }
127:           # Since the current requirement conflicts with an activated gem, we need
128:           # to backtrack to the current requirement's parent and try another version
129:           # of it (maybe the current requirement won't be present anymore). If the
130:           # current requirement is a root level requirement, we need to jump back to
131:           # where the conflicting gem was activated.
132:           parent = current.required_by.last || existing.required_by.last
133:           # We track the spot where the current gem was activated because we need
134:           # to keep a list of every spot a failure happened.
135:           debug { "    -> Jumping to: #{parent.name}" }
136:           throw parent.name, existing.required_by.last.name
137:         end
138:       else
139:         # There are no activated gems for the current requirement, so we are going
140:         # to find all gems that match the current requirement and try them in decending
141:         # order. We also need to keep a set of all conflicts that happen while trying
142:         # this gem. This is so that if no versions work, we can figure out the best
143:         # place to backtrack to.
144:         conflicts = Set.new
145: 
146:         # Fetch all gem versions matching the requirement
147:         #
148:         # TODO: Warn / error when no matching versions are found.
149:         matching_versions = search(current)
150: 
151:         if matching_versions.empty?
152:           if current.required_by.empty?
153:             if current.source
154:               name = current.name
155:               versions = @source_requirements[name][name].map { |s| s.version }
156:               message  = "Could not find gem '#{current}' in #{current.source}.\n"
157:               if versions.any?
158:                 message << "Source contains '#{name}' at: #{versions.join(', ')}"
159:               else
160:                 message << "Source does not contain any versions of '#{current}'"
161:               end
162:             else
163:               message = "Could not find gem '#{current}' "
164:               if @index.sources.include?(Bundler::Source::Rubygems)
165:                 message << "in any of the gem sources."
166:               else
167:                 message << "in the gems available on this machine."
168:               end
169:             end
170:             raise GemNotFound, message
171:           else
172:             @errors[current.name] = [nil, current]
173:           end
174:         end
175: 
176:         matching_versions.reverse_each do |spec|
177:           conflict = resolve_requirement(spec, current, reqs.dup, activated.dup)
178:           conflicts << conflict if conflict
179:         end
180:         # If the current requirement is a root level gem and we have conflicts, we
181:         # can figure out the best spot to backtrack to.
182:         if current.required_by.empty? && !conflicts.empty?
183:           # Check the current "catch" stack for the first one that is included in the
184:           # conflicts set. That is where the parent of the conflicting gem was required.
185:           # By jumping back to this spot, we can try other version of the parent of
186:           # the conflicting gem, hopefully finding a combination that activates correctly.
187:           @stack.reverse_each do |savepoint|
188:             if conflicts.include?(savepoint)
189:               debug { "    -> Jumping to: #{savepoint}" }
190:               throw savepoint
191:             end
192:           end
193:         end
194:       end
195:     end

197:     def resolve_requirement(spec, requirement, reqs, activated)
198:       # We are going to try activating the spec. We need to keep track of stack of
199:       # requirements that got us to the point of activating this gem.
200:       spec.required_by.replace requirement.required_by
201:       spec.required_by << requirement
202: 
203:       activated[spec.name] = spec
204:       debug { "  Activating: #{spec.name} (#{spec.version})" }
205:       debug { spec.required_by.map { |d| "    * #{d.name} (#{d.requirement})" }.join("\n") }
206: 
207:       # Now, we have to loop through all child dependencies and add them to our
208:       # array of requirements.
209:       debug { "    Dependencies"}
210:       spec.dependencies.each do |dep|
211:         next if dep.type == :development
212:         debug { "    * #{dep.name} (#{dep.requirement})" }
213:         dep.required_by.replace(requirement.required_by)
214:         dep.required_by << requirement
215:         reqs << dep
216:       end
217: 
218:       # We create a savepoint and mark it by the name of the requirement that caused
219:       # the gem to be activated. If the activated gem ever conflicts, we are able to
220:       # jump back to this point and try another version of the gem.
221:       length = @stack.length
222:       @stack << requirement.name
223:       retval = catch(requirement.name) do
224:         resolve(reqs, activated)
225:       end
226:       # Since we're doing a lot of throw / catches. A push does not necessarily match
227:       # up to a pop. So, we simply slice the stack back to what it was before the catch
228:       # block.
229:       @stack.slice!(length..1)
230:       retval
231:     end
